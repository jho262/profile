MYSQL DATABASE - GETTING STARTED

—— The below example commands were compiled from
http://dev.mysql.com/doc/mysql-getting-started/en/

— Creating a database and tables

CREATE DATABASE pets;

SHOW DATABASES;

USE pets;

CREATE TABLE cats
(
id INT unsigned NOT NULL AUTO_INCREMENT,
name VARCHAR(150) NOT NULL,
owner VARCHAR(150) NOT NULL,
birth DATE NOT NULL,
PRIMARY KEY (id)
);


SHOW TABLES;

DESCRIBE cats;

INSERT INTO cats ( name, owner, birth) VALUES
( ‘Sandy’, ‘Lennon’, ‘2015-01-03’ ),
( ‘Cookie’, ‘Casey’, ‘2013-11-13’ ),
( ‘Charlie’, ‘River’, ‘2016-05-21’ );

SELECT * FROM cats;

DELETE FROM cats where name=‘Cookie’;

ALTER TABLE cats ADD gender CHAR(1) AFTER name;

SHOW CREATE TABLE cats; — shows SQL that creates table

ALTER TABLE cats DROP gender;



—— Creating users

CREATE USER ‘jeffrey’@‘localhost’
IDENTIFIED BY ‘new_password’
PASSWORD EXPIRE; — requires user to set new password at 1st login

GRANT ALL ON pets.* TO ‘jeffrey’@‘localhost’;
GRANT SELECT ON db2.invoice TO ‘jeffrey’@‘localhost’;
ALTER USER ‘jeffrey’@‘localhost’ WITH MAX_QUERIES_PER_HOUR 90;
