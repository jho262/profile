HADOOP SAMPLE DFS COMMANDS

NOTE: hdfs://localhost:9000 is defined in $HADOOP_PREFIX/etc/hadoop/core-site.xml

Before executing the below commands, first execute "cd $HADOOP_PREFIX"

bin/hdfs dfs -mkdir -p hdfs://localhost:9000/input_dir/level1/level2/level3

bin/hdfs dfs -mkdir hdfs://localhost:9000/user/jamesho

bin/hdfs dfs -put etc/hadoop hdfs://localhost:9000/

bin/hdfs dfs -put htest.txt hdfs://localhost:9000/user/jamesho

bin/hdfs dfs -cp hdfs://localhost:9000/user/jamesho/htest.txt hdfs://localhost:9000/input_dir

bin/hdfs dfs -cat hdfs://localhost:9000/input_dir/htest.txt

bin/hdfs dfs -appendToFile htest2.txt hdfs://localhost:9000/input_dir/htest.txt

bin/hdfs dfs -checksum hdfs://localhost:9000/input_dir/htest.txt

bin/hdfs dfs -df hdfs://localhost:9000/input_dir

bin/hdfs dfs -du -s hdfs://localhost:9000/input_dir

bin/hdfs dfs -du hdfs://localhost:9000/input_dir

bin/hdfs dfs -find hdfs://localhost:9000/

bin/hdfs dfs -find hdfs://localhost:9000/input_dir

bin/hdfs dfs -get hdfs://localhost:9000/input_dir/htest.txt hget.txt

bin/hdfs dfs -getfacl hdfs://localhost:9000/input_dir

bin/hdfs dfs -getfacl hdfs://localhost:9000/input_dir/htest.txt

bin/hdfs dfs -ls hdfs://localhost:9000/

bin/hdfs dfs -stat hdfs://localhost:9000/input_dir/htest.txt

bin/hdfs dfs -test -z hdfs://localhost:9000/input_dir/htest.txt

bin/hdfs dfs -rm hdfs://localhost:9000/Users

bin/hdfs dfs -rm hdfs://localhost:9000/user/jamesho/htest.txt

bin/hdfs dfs -rmdir hdfs://localhost:9000/Users

bin/hdfs dfs -touchz hdfs://localhost:9000/input_dir/level1/level2/level3/t3.txt
